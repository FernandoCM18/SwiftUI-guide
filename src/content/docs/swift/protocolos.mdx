## ¿Qué son los protocolos?

Un protocolo define un contrato de métodos, propiedades y otros requisitos que un tipo debe implementar. Es similar a las interfaces en otros lenguajes, pero más potente gracias a las extensiones de protocolo.

## Declaración de un protocolo

```swift
protocol Describible {
    var descripcion: String { get }
    func describir()
}

protocol Identificable {
    var id: String { get set }
}
```

## Conformidad a un protocolo

```swift
protocol Describible {
    var descripcion: String { get }
    func describir()
}

struct Producto: Describible {
    var nombre: String
    var precio: Double

    // Implementación requerida
    var descripcion: String {
        return "\(nombre) - \(precio)€"
    }

    func describir() {
        print(descripcion)
    }
}

let producto = Producto(nombre: "Teclado", precio: 49.99)
producto.describir()  // "Teclado - 49.99€"
```

## Protocolos con métodos

```swift
protocol Calculable {
    func area() -> Double
    func perimetro() -> Double
}

struct Rectangulo: Calculable {
    var ancho: Double
    var alto: Double

    func area() -> Double {
        return ancho * alto
    }

    func perimetro() -> Double {
        return 2 * (ancho + alto)
    }
}

struct Circulo: Calculable {
    var radio: Double

    func area() -> Double {
        return .pi * radio * radio
    }

    func perimetro() -> Double {
        return 2 * .pi * radio
    }
}

let formas: [Calculable] = [
    Rectangulo(ancho: 5, alto: 3),
    Circulo(radio: 4)
]

for forma in formas {
    print("Área: \(forma.area()), Perímetro: \(forma.perimetro())")
}
```

## Protocolos con métodos mutating

Para permitir que los structs modifiquen sus propiedades.

```swift
protocol Conmutable {
    mutating func conmutar()
}

struct Interruptor: Conmutable {
    var encendido = false

    mutating func conmutar() {
        encendido.toggle()
    }
}

var luz = Interruptor()
print("Encendido: \(luz.encendido)")  // false
luz.conmutar()
print("Encendido: \(luz.encendido)")  // true
```

## Herencia de protocolos

```swift
protocol Nombrable {
    var nombre: String { get }
}

protocol Saludable: Nombrable {
    func saludar() -> String
}

struct Usuario: Saludable {
    var nombre: String

    func saludar() -> String {
        return "¡Hola! Soy \(nombre)"
    }
}

let usuario = Usuario(nombre: "Ana")
print(usuario.saludar())
```

## Composición de protocolos

Puedes requerir que un tipo conforme a múltiples protocolos.

```swift
protocol Nombrable {
    var nombre: String { get }
}

protocol ConEdad {
    var edad: Int { get }
}

// El parámetro debe conformar AMBOS protocolos
func presentar(_ persona: Nombrable & ConEdad) {
    print("\(persona.nombre), \(persona.edad) años")
}

struct Empleado: Nombrable, ConEdad {
    var nombre: String
    var edad: Int
}

let empleado = Empleado(nombre: "Carlos", edad: 30)
presentar(empleado)
```

## Extensiones de protocolo

Las extensiones permiten proporcionar implementaciones por defecto.

```swift
protocol Saludable {
    var nombre: String { get }
    func saludar() -> String
}

// Implementación por defecto
extension Saludable {
    func saludar() -> String {
        return "¡Hola! Soy \(nombre)"
    }

    func despedir() -> String {
        return "¡Adiós de parte de \(nombre)!"
    }
}

struct Robot: Saludable {
    var nombre: String
    // No necesita implementar saludar(), usa el valor por defecto
}

struct Humano: Saludable {
    var nombre: String

    // Puede sobreescribir la implementación por defecto
    func saludar() -> String {
        return "¡Buenos días! Me llamo \(nombre)"
    }
}

let robot = Robot(nombre: "R2D2")
let humano = Humano(nombre: "María")

print(robot.saludar())    // "¡Hola! Soy R2D2"
print(humano.saludar())   // "¡Buenos días! Me llamo María"
print(robot.despedir())   // "¡Adiós de parte de R2D2!"
```

> [Probar en Swift Playground →](https://swiftfiddle.com/)

:::tip
Las extensiones de protocolo son el corazón de la programación orientada a protocolos (POP) en Swift. Te permiten agregar comportamiento compartido sin herencia.
:::

## Programación orientada a protocolos

Un ejemplo práctico que muestra cómo los protocolos superan a la herencia clásica.

```swift
// Definir comportamientos como protocolos
protocol Volador {
    func volar() -> String
}

protocol Nadador {
    func nadar() -> String
}

protocol Corredor {
    func correr() -> String
}

// Implementaciones por defecto
extension Volador {
    func volar() -> String { "Estoy volando" }
}

extension Nadador {
    func nadar() -> String { "Estoy nadando" }
}

extension Corredor {
    func correr() -> String { "Estoy corriendo" }
}

// Componer comportamientos libremente
struct Pato: Volador, Nadador, Corredor {
    var nombre: String
}

struct Pinguino: Nadador, Corredor {
    var nombre: String
}

struct Aguila: Volador {
    var nombre: String
}

let pato = Pato(nombre: "Donald")
print(pato.volar())   // "Estoy volando"
print(pato.nadar())   // "Estoy nadando"
print(pato.correr())  // "Estoy corriendo"

let pinguino = Pinguino(nombre: "Tux")
print(pinguino.nadar())   // "Estoy nadando"
// pinguino.volar()  // Error: Pinguino no conforma Volador
```

## Protocolos estándar de Swift

Swift define muchos protocolos que puedes adoptar.

```swift
// Equatable: permite comparar con ==
struct Moneda: Equatable {
    var valor: Double
    var codigo: String
}

let euro1 = Moneda(valor: 1.0, codigo: "EUR")
let euro2 = Moneda(valor: 1.0, codigo: "EUR")
print(euro1 == euro2)  // true

// Comparable: permite ordenar con <, >, etc.
struct Puntuacion: Comparable {
    var valor: Int
    var nombre: String

    static func < (lhs: Puntuacion, rhs: Puntuacion) -> Bool {
        return lhs.valor < rhs.valor
    }
}

let puntuaciones = [
    Puntuacion(valor: 85, nombre: "Ana"),
    Puntuacion(valor: 92, nombre: "Luis"),
    Puntuacion(valor: 78, nombre: "María")
]
let ordenadas = puntuaciones.sorted()
for p in ordenadas {
    print("\(p.nombre): \(p.valor)")
}

// CustomStringConvertible: personalizar print()
struct Libro: CustomStringConvertible {
    var titulo: String
    var autor: String

    var description: String {
        return "\"\(titulo)\" por \(autor)"
    }
}

let libro = Libro(titulo: "Don Quijote", autor: "Cervantes")
print(libro)  // "Don Quijote" por Cervantes
```

## Resumen

| Concepto | Descripción |
|----------|-------------|
| Protocolo | Contrato que define requisitos |
| Conformidad | Un tipo adopta el protocolo e implementa sus requisitos |
| Extensión de protocolo | Implementaciones por defecto |
| Composición | `Protocolo1 & Protocolo2` |
| POP | Componer comportamiento con protocolos en lugar de herencia |
