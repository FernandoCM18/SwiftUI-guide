## El componente Label

`Label` combina un texto con un icono, creando un elemento visual consistente ideal para menús, listas y botones. Utiliza SF Symbols para los iconos del sistema.

![Label](../../../../assets/swiftui/label.svg)

## Uso básico

<Tabs syncKey="lang">
  <TabItem label="Swift">
    ```swift
    import SwiftUI

    struct LabelBasico: View {
        var body: some View {
            VStack(alignment: .leading, spacing: 16) {
                Label("Favoritos", systemImage: "star.fill")
                Label("Compartir", systemImage: "square.and.arrow.up")
    ```

## Label con estilos

SwiftUI proporciona varios estilos predefinidos para Labels.

<Tabs syncKey="lang">
  <TabItem label="Swift">
    ```swift
    import SwiftUI

    struct LabelEstilos: View {
        var body: some View {
            VStack(spacing: 20) {
                // Estilo por defecto: icono + texto
                Label("Automático", systemImage: "wand.and.stars")
                    .labelStyle(.automatic)

                // Solo icono
                Label("Solo icono", systemImage: "heart.fill")
                    .labelStyle(.iconOnly)

                // Solo título
                Label("Solo título", systemImage: "heart.fill")
                    .labelStyle(.titleOnly)

                // Título y icono (explícito)
                Label("Ambos", systemImage: "bolt.fill")
                    .labelStyle(.titleAndIcon)
            }
            .font(.title3)
        }
    }
    ```
  </TabItem>
  <TabItem label="React (Next.js)">
    ```tsx
    import { Sparkles, Heart, Zap } from "lucide-react";

    export default function LabelEstilos() {
      return (
        <div className="flex flex-col gap-5 text-lg">
          {/* Icono + texto (por defecto) */}
          <span className="inline-flex items-center gap-2">
            <Sparkles className="size-5" /> Automático
          </span>
          {/* Solo icono */}
          <span>
            <Heart className="size-5" />
          </span>
          {/* Solo título */}
          <span>Solo título</span>
          {/* Título y icono */}
          <span className="inline-flex items-center gap-2">
            <Zap className="size-5" /> Ambos
          </span>
        </div>
      );
    }
    ```
  </TabItem>
</Tabs>

## Label con contenido personalizado

Puedes crear un Label con vistas personalizadas para el título y el icono.

<Tabs syncKey="lang">
  <TabItem label="Swift">
    ```swift
    import SwiftUI

    struct LabelCustom: View {
        var body: some View {
            VStack(spacing: 20) {
                // Label con vistas personalizadas
                Label {
                    Text("Descarga completada")
                        .font(.headline)
                } icon: {
                    Image(systemName: "arrow.down.circle.fill")
                        .foregroundStyle(.green)
                        .font(.title)
                }

                // Label con múltiples líneas de texto
                Label {
                    VStack(alignment: .leading) {
                        Text("María García")
                            .font(.headline)
                        Text("Desarrolladora iOS")
                            .font(.caption)
                            .foregroundStyle(.secondary)
                    }
                } icon: {
                    Image(systemName: "person.crop.circle.fill")
                        .font(.largeTitle)
                        .foregroundStyle(.blue)
                }
            }
        }
    }
    ```
  </TabItem>
  <TabItem label="React (Next.js)">

    ```tsx
          </div>
    ```

## Label como botón

<Tabs syncKey="lang">
  <TabItem label="Swift">
    ```swift
    import SwiftUI

    struct LabelBoton: View {
        @State private var esFavorito = false

        var body: some View {
            VStack(spacing: 20) {
                Button {
                    esFavorito.toggle()
                } label: {
                    Label(
                        esFavorito ? "Quitar de favoritos" : "Añadir a favoritos",
                        systemImage: esFavorito ? "star.fill" : "star"
                    )
                    .foregroundStyle(esFavorito ? .yellow : .gray)
                }

                Button {
                    // Acción de compartir
                } label: {
                    Label("Compartir", systemImage: "square.and.arrow.up")
                        .padding()
                        .background(Color.blue)
                        .foregroundStyle(.white)
                        .clipShape(.rect(cornerRadius: 10))
                }
            }
        }
    }
    ```
  </TabItem>
  <TabItem label="React (Next.js)">
    ```tsx
    "use client";

    import { useState } from "react";
    import { Star, Share2 } from "lucide-react";

    export default function LabelBoton() {
      const [esFavorito, setEsFavorito] = useState(false);

      return (
        <div className="flex flex-col items-center gap-5">
          <button
            onClick={() => setEsFavorito(!esFavorito)}
            className={`inline-flex items-center gap-2 ${
              esFavorito ? "text-yellow-500" : "text-gray-400"
            }`}
          >
            <Star
              className="size-5"
              fill={esFavorito ? "currentColor" : "none"}
            />
            {esFavorito ? "Quitar de favoritos" : "Añadir a favoritos"}
          </button>
          <button className="inline-flex items-center gap-2 rounded-lg bg-blue-500 px-4 py-2 text-white">
            <Share2 className="size-5" /> Compartir
          </button>
        </div>
      );
    }
    ```
  </TabItem>
</Tabs>

## Resumen

| Propiedad | Descripción |
|-----------|-------------|
| `Label("texto", systemImage:)` | Crea un label con icono del sistema |
| `.labelStyle(.iconOnly)` | Muestra solo el icono |
| `.labelStyle(.titleOnly)` | Muestra solo el texto |
| `.labelStyle(.titleAndIcon)` | Muestra ambos explícitamente |
| `Label { } icon: { }` | Label con contenido personalizado |
