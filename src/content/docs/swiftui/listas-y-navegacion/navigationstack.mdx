`NavigationStack` proporciona una pila de navegación que permite al usuario ir de una vista a otra con transiciones animadas y un botón de retroceso automático.

## Vista previa

![NavigationStack en SwiftUI](../../../../assets/swiftui/navigationstack.svg)

## Uso básico

<Tabs syncKey="lang">
  <TabItem label="Swift">
    ```swift
    NavigationStack {

                Text("Vista de Perfil")
            }
            NavigationLink("Configuración") {
                Text("Vista de Configuración")
            }
        }
        .navigationTitle("Inicio")
    }
    ```
  </TabItem>
  <TabItem label="React">

## NavigationLink con datos

<Tabs syncKey="lang">
  <TabItem label="Swift">
    ```swift
    struct Fruta: Identifiable, Hashable {
        let id = UUID()
        let nombre: String
        let emoji: String
    }

    struct FrutasView: View {
        let frutas = [
            Fruta(nombre: "Manzana", emoji: "🍎"),
            Fruta(nombre: "Plátano", emoji: "🍌"),
            Fruta(nombre: "Naranja", emoji: "🍊"),
        ]

        var body: some View {
            NavigationStack {
                List(frutas) { fruta in
                    NavigationLink(value: fruta) {
                        Text("\(fruta.emoji) \(fruta.nombre)")
                    }
                }
                .navigationTitle("Frutas")
                .navigationDestination(for: Fruta.self) { fruta in
                    Text("Detalle de \(fruta.nombre)")
                        .font(.largeTitle)
                }
            }
        }
    }
    ```
  </TabItem>
  <TabItem label="React">
    ```tsx
    // app/frutas/page.tsx
    import Link from "next/link";

    const frutas = [
      { id: "manzana", nombre: "Manzana", emoji: "🍎" },
      { id: "platano", nombre: "Plátano", emoji: "🍌" },
      { id: "naranja", nombre: "Naranja", emoji: "🍊" },
    ];

    export default function FrutasPage() {
      return (
        <div className="mx-auto max-w-md">
          <h1 className="mb-4 text-2xl font-bold">Frutas</h1>
          <ul className="divide-y divide-gray-200 rounded-lg bg-white">
            {frutas.map((fruta) => (
              <li key={fruta.id}>
                <Link
                  href={`/frutas/${fruta.id}`}
                  className="block px-4 py-3 hover:bg-gray-50"
                >
                  {fruta.emoji} {fruta.nombre}
                </Link>
              </li>
            ))}
          </ul>
        </div>
      );
    }

    // app/frutas/[id]/page.tsx
    export default function FrutaDetalle({ params }: { params: { id: string } }) {
      return (
        <h1 className="text-4xl font-bold">
          Detalle de {params.id}
        </h1>
      );
    }
    ```
  </TabItem>
</Tabs>

## Barra de herramientas

<Tabs syncKey="lang">
  <TabItem label="Swift">
    ```swift
    NavigationStack {
        Text("Contenido")
            .navigationTitle("Mi App")
            .toolbar {
                ToolbarItem(placement: .navigationBarTrailing) {
                    Button {
                        // Acción
                    } label: {
                        Image(systemName: "plus")
                    }
                }

                ToolbarItem(placement: .navigationBarLeading) {
                    Button("Editar") {
                        // Acción
                    }
                }
            }
    }
    ```
  </TabItem>
  <TabItem label="React">

    export default function MiAppLayout({ children }: { children: React.ReactNode }) {

    }

## Estilos de título

<Tabs syncKey="lang">
  <TabItem label="Swift">
    ```swift
    // Título grande (por defecto)
    .navigationTitle("Título")
    .navigationBarTitleDisplayMode(.large)

    // Título en línea (compacto)
    .navigationTitle("Título")
    .navigationBarTitleDisplayMode(.inline)
    ```
  </TabItem>
  <TabItem label="React">
    ```tsx
    // Título grande
    <h1 className="text-3xl font-bold">Título</h1>

    // Título compacto (en la barra de navegación)
    <nav className="border-b px-4 py-3">
      <h1 className="text-center text-lg font-semibold">Título</h1>
    </nav>
    ```
  </TabItem>
</Tabs>

:::tip
`NavigationStack` reemplaza a `NavigationView` desde iOS 16. Usa siempre `NavigationStack` en proyectos nuevos.
:::
