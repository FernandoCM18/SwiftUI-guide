El componente `Slider` permite al usuario seleccionar un valor dentro de un rango definido arrastrando un indicador.

## Vista previa

![Slider en SwiftUI](../../../../assets/swiftui/slider.svg)

## Uso básico

<Tabs syncKey="lang">
  <TabItem label="Swift">
    ```swift
    struct SliderEjemplo: View {
        @State private var valor = 50.0

        var body: some View {
            VStack {
                Slider(value: $valor, in: 0...100)
                Text("Valor: \(valor, specifier: "%.0f")")
            }
            .padding()
        }
    }
    ```
  </TabItem>
  <TabItem label="React">
    ```tsx
    "use client";

    import { useState } from "react";

    export default function SliderEjemplo() {
      const [valor, setValor] = useState(50);

      return (
        <div className="flex flex-col gap-2 p-4">
          <input
            type="range"
            min={0}
            max={100}
            value={valor}
            onChange={(e) => setValor(Number(e.target.value))}
            className="w-full accent-blue-500"
          />
          <p className="text-center">Valor: {valor}</p>
        </div>
      );
    }
    ```
  </TabItem>
</Tabs>

## Slider con paso definido

<Tabs syncKey="lang">
  <TabItem label="Swift">
    ```swift
    Slider(value: $temperatura, in: 16...30, step: 0.5) {
        Text("Temperatura")

    } maximumValueLabel: {
        Text("30°")
    ```

  </TabItem>
  <TabItem label="React">
    ```tsx

    import { useState } from "react";

    export default function SliderTemperatura() {
      const [temperatura, setTemperatura] = useState(23);

      return (
        <div className="flex items-center gap-3 p-4">
          <span className="text-sm text-gray-500">16°</span>
          <input
            type="range"
            min={16}
            max={30}
            step={0.5}
            value={temperatura}
            onChange={(e) => setTemperatura(Number(e.target.value))}
            className="flex-1 accent-blue-500"
          />
          <span className="text-sm text-gray-500">30°</span>
        </div>
      );
    }
    ```

## Detectar cambios

<Tabs syncKey="lang">
  <TabItem label="Swift">
    ```swift
    Slider(value: $brillo, in: 0...1) {
        Text("Brillo")
    } onEditingChanged: { editando in
        if !editando {
            // El usuario soltó el slider
            guardarBrillo(brillo)
        }
    }
    ```
  </TabItem>
  <TabItem label="React">
    ```tsx
    "use client";

    import { useState } from "react";

    function guardarBrillo(valor: number) {
      console.log("Brillo guardado:", valor);
    }

    export default function SliderBrillo() {
      const [brillo, setBrillo] = useState(0.5);

      return (
        <input
          type="range"
          min={0}
          max={1}
          step={0.01}
          value={brillo}
          onChange={(e) => setBrillo(Number(e.target.value))}
          onMouseUp={() => guardarBrillo(brillo)}
          onTouchEnd={() => guardarBrillo(brillo)}
          className="w-full accent-blue-500 p-4"
        />
      );
    }
    ```
  </TabItem>
</Tabs>

## Modificadores comunes

| Modificador | Descripción |
|---|---|
| `.tint(.blue)` | Color de la pista del slider |
| `.disabled(true)` | Deshabilita el slider |
| `step:` | Define el incremento entre valores |

:::tip
Combina `Slider` con `Text` para mostrar el valor actual al usuario.
:::

## Ejemplo completo

<Tabs syncKey="lang">
  <TabItem label="Swift">
    ```swift
    struct ControlVolumenView: View {
        @State private var volumen = 0.5
        @State private var brillo = 0.8
        @State private var velocidad = 1.0

        var body: some View {
            Form {
                Section("Audio") {
                    HStack {
                        Image(systemName: "speaker.fill")
                        Slider(value: $volumen, in: 0...1)
                        Image(systemName: "speaker.wave.3.fill")
                    }

                    Text("Volumen: \(Int(volumen * 100))%")
                        .foregroundStyle(.secondary)
                }

                Section("Pantalla") {
                    HStack {
                        Image(systemName: "sun.min")
                        Slider(value: $brillo, in: 0...1)
                        Image(systemName: "sun.max")
                    }
                    .tint(.yellow)
                }

                Section("Reproducción") {
                    Slider(value: $velocidad, in: 0.5...2.0, step: 0.25) {
                        Text("Velocidad")
                    } minimumValueLabel: {
                        Text("0.5x")
                    } maximumValueLabel: {
                        Text("2x")
                    }

                    Text("Velocidad: \(velocidad, specifier: "%.2f")x")
                        .foregroundStyle(.secondary)
                }
            }
        }
    }
    ```
