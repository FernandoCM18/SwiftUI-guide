Conectar una app SwiftUI a una API REST es una de las tareas más comunes. Swift ofrece herramientas nativas potentes: `URLSession`, `async/await` y `Codable`.

## Flujo completo

```
API REST → URLSession → JSON → Codable (decode) → Model → ViewModel → View
```

## 1. Definir el Model

Usa `Codable` para mapear JSON a structs Swift automáticamente:

<Tabs syncKey="lang">
  <TabItem label="Swift">
    ```swift
    // JSON de la API:
    // { "id": 1, "name": "iPhone 16", "price": 999.99, "in_stock": true }

    struct Producto: Codable, Identifiable {
        let id: Int
        let name: String
        let price: Double
        let inStock: Bool

        // Mapear snake_case del JSON a camelCase de Swift
        enum CodingKeys: String, CodingKey {
            case id, name, price
            case inStock = "in_stock"
        }
    }
    ```

## 2. Crear el servicio de red

<Tabs syncKey="lang">
  <TabItem label="Swift">
    ```swift
    enum NetworkError: Error, LocalizedError {
        case urlInvalida
        case sinRespuesta
        case decodificacionFallida
        case serverError(Int)

        var errorDescription: String? {
            switch self {
            case .urlInvalida: return "URL inválida"
            case .sinRespuesta: return "No se recibió respuesta del servidor"
            case .decodificacionFallida: return "Error al procesar los datos"
            case .serverError(let code): return "Error del servidor: \(code)"
            }
        }
    }

    class APIService {
        static let shared = APIService()
        private let baseURL = "https://api.ejemplo.com"

        func fetch<T: Decodable>(_ endpoint: String) async throws -> T {
            guard let url = URL(string: "\(baseURL)\(endpoint)") else {
                throw NetworkError.urlInvalida
            }

            let (data, response) = try await URLSession.shared.data(from: url)

            guard let httpResponse = response as? HTTPURLResponse else {
                throw NetworkError.sinRespuesta
            }

            guard (200...299).contains(httpResponse.statusCode) else {
                throw NetworkError.serverError(httpResponse.statusCode)
            }

            let decoder = JSONDecoder()
            decoder.keyDecodingStrategy = .convertFromSnakeCase

            do {
                return try decoder.decode(T.self, from: data)
            } catch {
                throw NetworkError.decodificacionFallida
            }
        }
    }
    ```
  </TabItem>
  <TabItem label="React">
    ```typescript
    class NetworkError extends Error {
      statusCode?: number;
      constructor(message: string, statusCode?: number) {
        super(message);
        this.statusCode = statusCode;
      }
    }

    const BASE_URL = "https://api.ejemplo.com";

    async function fetchAPI<T>(endpoint: string): Promise<T> {
      const res = await fetch(`${BASE_URL}${endpoint}`);

      if (!res.ok) {
        throw new NetworkError(
          `Error del servidor: ${res.status}`,
          res.status
        );
      }

      return res.json();
    }
    ```
  </TabItem>
</Tabs>

## 3. ViewModel con async/await

<Tabs syncKey="lang">
  <TabItem label="Swift">
    ```swift
    import SwiftUI

    @Observable
    class ProductosViewModel {
        var productos: [Producto] = []
        var estaCargando = false
        var mensajeError: String?

        func cargarProductos() async {
            estaCargando = true
            mensajeError = nil

            do {
                productos = try await APIService.shared.fetch("/productos")
            } catch {
                mensajeError = error.localizedDescription
            }

            estaCargando = false
        }

        func refrescar() async {
            await cargarProductos()
        }
    }
    ```
  </TabItem>
  <TabItem label="React">
    ```tsx
    "use client";

    import { useState, useEffect } from "react";

    function useProductos() {
      const [productos, setProductos] = useState<Producto[]>([]);
      const [estaCargando, setEstaCargando] = useState(false);
      const [mensajeError, setMensajeError] = useState<string | null>(null);

      async function cargarProductos() {
        setEstaCargando(true);
        setMensajeError(null);

        try {
          const data = await fetchAPI<Producto[]>("/productos");
          setProductos(data);
        } catch (error) {
          setMensajeError(
            error instanceof Error ? error.message : "Error desconocido"
          );
        } finally {
          setEstaCargando(false);
        }
      }

      useEffect(() => {
        cargarProductos();
      }, []);

      return { productos, estaCargando, mensajeError, refrescar: cargarProductos };
    }
    ```

## Enviar datos (POST)

<Tabs syncKey="lang">
  <TabItem label="Swift">
    ```swift
    extension APIService {
        func post<T: Codable, R: Decodable>(
            _ endpoint: String,
            body: T
        ) async throws -> R {
            guard let url = URL(string: "\(baseURL)\(endpoint)") else {
                throw NetworkError.urlInvalida
            }

            var request = URLRequest(url: url)
            request.httpMethod = "POST"
            request.setValue("application/json", forHTTPHeaderField: "Content-Type")

            let encoder = JSONEncoder()
            encoder.keyEncodingStrategy = .convertToSnakeCase
            request.httpBody = try encoder.encode(body)

            let (data, response) = try await URLSession.shared.data(for: request)

            guard let httpResponse = response as? HTTPURLResponse,
                  (200...299).contains(httpResponse.statusCode) else {
                throw NetworkError.sinRespuesta
            }

            return try JSONDecoder().decode(R.self, from: data)
        }
    }

    // Uso:
    struct NuevoProducto: Codable {
        let name: String
        let price: Double
    }

    let nuevo = NuevoProducto(name: "iPad", price: 799)
    let creado: Producto = try await APIService.shared.post("/productos", body: nuevo)
    ```
  </TabItem>
  <TabItem label="React">
    ```typescript
    async function postAPI<T, R>(endpoint: string, body: T): Promise<R> {
      const res = await fetch(`${BASE_URL}${endpoint}`, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(body),
      });

      if (!res.ok) {
        throw new NetworkError(`Error del servidor: ${res.status}`, res.status);
      }

      return res.json();
    }

    // Uso:
    interface NuevoProducto {
      name: string;
      price: number;
    }

    const nuevo: NuevoProducto = { name: "iPad", price: 799 };
    const creado = await postAPI<NuevoProducto, Producto>("/productos", nuevo);
    ```
  </TabItem>
</Tabs>

## Autenticación con Bearer Token

<Tabs syncKey="lang">
  <TabItem label="Swift">
    ```swift
    extension APIService {
        func fetchAutenticado<T: Decodable>(
            _ endpoint: String,
            token: String
        ) async throws -> T {
            guard let url = URL(string: "\(baseURL)\(endpoint)") else {
                throw NetworkError.urlInvalida
            }

            var request = URLRequest(url: url)
            request.setValue("Bearer \(token)", forHTTPHeaderField: "Authorization")

            let (data, _) = try await URLSession.shared.data(for: request)
            return try JSONDecoder().decode(T.self, from: data)
        }
    }
    ```
  </TabItem>
  <TabItem label="React">
    ```typescript
    async function fetchAutenticado<T>(
      endpoint: string,
      token: string
    ): Promise<T> {
      const res = await fetch(`${BASE_URL}${endpoint}`, {
        headers: { Authorization: `Bearer ${token}` },
      });

      if (!res.ok) {
        throw new NetworkError(`Error: ${res.status}`, res.status);
      }

      return res.json();
    }
    ```
  </TabItem>

## Ejemplo real: App del clima

<Tabs syncKey="lang">
  <TabItem label="Swift">
    ```swift
    // Model
    struct Clima: Codable {
        let temperatura: Double
        let descripcion: String
        let icono: String

        enum CodingKeys: String, CodingKey {
            case temperatura = "temp"
            case descripcion = "description"
            case icono = "icon"
        }
    }

    // ViewModel
    @Observable
    class ClimaViewModel {
        var clima: Clima?
        var estaCargando = false
        var error: String?

        func obtenerClima(ciudad: String) async {
            estaCargando = true
            error = nil

            do {
                let endpoint = "/clima?ciudad=\(ciudad.addingPercentEncoding(withAllowedCharacters: .urlQueryAllowed) ?? ciudad)"
                clima = try await APIService.shared.fetch(endpoint)
            } catch {
                self.error = error.localizedDescription
            }

            estaCargando = false
        }
    }

    // View
    struct ClimaView: View {
        @State private var vm = ClimaViewModel()
        @State private var ciudad = "Madrid"

        var body: some View {
            VStack(spacing: 20) {
                TextField("Ciudad", text: $ciudad)
                    .textFieldStyle(.roundedBorder)
                    .submitLabel(.search)
                    .onSubmit {
                        Task { await vm.obtenerClima(ciudad: ciudad) }
                    }

                if vm.estaCargando {
                    ProgressView()
                } else if let clima = vm.clima {
                    VStack {
                        Text("\(clima.temperatura, specifier: "%.1f")°C")
                            .font(.system(size: 64, weight: .thin))
                        Text(clima.descripcion)
                            .font(.title3)
                            .foregroundStyle(.secondary)
                    }
                } else if let error = vm.error {
                    Text(error)
                        .foregroundStyle(.red)
                }
            }
            .padding()
            .task { await vm.obtenerClima(ciudad: ciudad) }
        }
    }
    ```
  </TabItem>
  <TabItem label="React">
    ```tsx
    "use client";

    import { useState, useEffect } from "react";

    // Model
    interface Clima {
      temp: number;
      description: string;
      icon: string;
    }

    // Custom Hook (equivalente al ViewModel)
    function useClima(ciudadInicial: string) {
      const [clima, setClima] = useState<Clima | null>(null);
      const [estaCargando, setEstaCargando] = useState(false);
      const [error, setError] = useState<string | null>(null);

      async function obtenerClima(ciudad: string) {
        setEstaCargando(true);
        setError(null);

        try {
          const data = await fetchAPI<Clima>(
            `/clima?ciudad=${encodeURIComponent(ciudad)}`
          );
          setClima(data);
        } catch (err) {
          setError(
            err instanceof Error ? err.message : "Error desconocido"
          );
        } finally {
          setEstaCargando(false);
        }
      }

      useEffect(() => {
        obtenerClima(ciudadInicial);
      }, [ciudadInicial]);

      return { clima, estaCargando, error, obtenerClima };
    }

    // Componente (equivalente a la View)
    export default function ClimaPage() {
      const [ciudad, setCiudad] = useState("Madrid");
      const { clima, estaCargando, error, obtenerClima } = useClima(ciudad);

      return (
        <div className="mx-auto max-w-md space-y-5 p-6">
          <form
            onSubmit={(e) => {
              e.preventDefault();
              obtenerClima(ciudad);
            }}
          >
            <input
              value={ciudad}
              onChange={(e) => setCiudad(e.target.value)}
              placeholder="Ciudad"
              className="w-full rounded-lg border px-4 py-2"
            />
          </form>

          {estaCargando && (
            <p className="text-center text-gray-500">Cargando...</p>
          )}

          {clima && !estaCargando && (
            <div className="text-center">
              <p className="text-6xl font-thin">{clima.temp.toFixed(1)}°C</p>
              <p className="mt-2 text-gray-500">{clima.description}</p>
            </div>
          )}

          {error && <p className="text-center text-red-500">{error}</p>}
        </div>
      );
    }
    ```
  </TabItem>
