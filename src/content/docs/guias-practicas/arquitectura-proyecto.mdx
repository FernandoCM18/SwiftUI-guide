## Convenciones de nombrado

### Archivos y tipos

## Patrones de arquitectura

### MVVM (recomendado para SwiftUI)

<Tabs syncKey="lang">
  <TabItem label="Swift">
    ```swift
    // Model: datos puros
    struct Tarea: Identifiable {
        let id = UUID()
        var titulo: String
        var completada: Bool
    }

    // ViewModel: lógica de negocio
    @Observable
    class TareasViewModel {
        var tareas: [Tarea] = []

        var pendientes: [Tarea] {
            tareas.filter { !$0.completada }
        }

        func agregar(_ titulo: String) {
            tareas.append(Tarea(titulo: titulo, completada: false))
        }

        func toggle(_ tarea: Tarea) {
            if let i = tareas.firstIndex(where: { $0.id == tarea.id }) {
                tareas[i].completada.toggle()
            }
        }
    }

    // View: solo UI
    struct TareasView: View {
        @State private var vm = TareasViewModel()

        var body: some View {
            List(vm.tareas) { tarea in
                Text(tarea.titulo)
            }
        }
    }
    ```
  </TabItem>
  <TabItem label="React">
    ```tsx
    "use client";

    import { useState } from "react";

    // Type: datos puros
    interface Tarea {
      id: string;
      titulo: string;
      completada: boolean;
    }

    // Custom Hook: lógica de negocio (reemplaza ViewModel)
    function useTareas() {
      const [tareas, setTareas] = useState<Tarea[]>([]);

      const pendientes = tareas.filter((t) => !t.completada);

      function agregar(titulo: string) {
        setTareas((prev) => [
          ...prev,
          { id: crypto.randomUUID(), titulo, completada: false },
        ]);
      }

      function toggle(id: string) {
        setTareas((prev) =>
          prev.map((t) =>
            t.id === id ? { ...t, completada: !t.completada } : t
          )
        );
      }

      return { tareas, pendientes, agregar, toggle };
    }

    // Componente: solo UI
    export default function TareasPage() {
      const { tareas, toggle } = useTareas();

      return (
        <ul>
          {tareas.map((tarea) => (
            <li key={tarea.id} onClick={() => toggle(tarea.id)}>
              {tarea.titulo}
            </li>
          ))}
        </ul>
      );
    }
    ```
  </TabItem>
</Tabs>

### Repository Pattern

Abstrae el origen de datos detrás de un protocolo:

<Tabs syncKey="lang">
  <TabItem label="Swift">
    ```swift
    // Protocolo
    protocol ProductoRepository {
        func obtenerTodos() async throws -> [Producto]
        func obtenerPorId(_ id: Int) async throws -> Producto
        func crear(_ producto: Producto) async throws -> Producto
    }

    // Implementación real (API)
    class APIProductoRepository: ProductoRepository {
        func obtenerTodos() async throws -> [Producto] {
            try await APIService.shared.fetch("/productos")
        }

        func obtenerPorId(_ id: Int) async throws -> Producto {
            try await APIService.shared.fetch("/productos/\(id)")
        }

        func crear(_ producto: Producto) async throws -> Producto {
            try await APIService.shared.post("/productos", body: producto)
        }
    }

    // Implementación mock (para previews y tests)
    class MockProductoRepository: ProductoRepository {
        func obtenerTodos() async throws -> [Producto] {
            [
                Producto(id: 1, name: "iPhone", price: 999, inStock: true),
                Producto(id: 2, name: "iPad", price: 799, inStock: false),
            ]
        }

        func obtenerPorId(_ id: Int) async throws -> Producto {
            Producto(id: id, name: "Test", price: 0, inStock: true)
        }

        func crear(_ producto: Producto) async throws -> Producto {
            producto
        }
    }
    ```
  </TabItem>
  <TabItem label="React">
    ```typescript
    // Interface (equivalente a Protocol)
    interface ProductoRepository {
      obtenerTodos(): Promise<Producto[]>;
      obtenerPorId(id: number): Promise<Producto>;
      crear(producto: Producto): Promise<Producto>;
    }

    // Implementación real (API)
    class APIProductoRepository implements ProductoRepository {
      async obtenerTodos() {
        return fetchAPI<Producto[]>("/productos");
      }

      async obtenerPorId(id: number) {
        return fetchAPI<Producto>(`/productos/${id}`);
      }

      async crear(producto: Producto) {
        return postAPI<Producto, Producto>("/productos", producto);
      }
    }

    // Implementación mock (para tests)
    class MockProductoRepository implements ProductoRepository {
      async obtenerTodos() {
        return [
          { id: 1, name: "iPhone", price: 999, inStock: true },
          { id: 2, name: "iPad", price: 799, inStock: false },
        ];
      }

      async obtenerPorId(id: number) {
        return { id, name: "Test", price: 0, inStock: true };
      }

      async crear(producto: Producto) {
        return producto;
      }
    }
    ```
  </TabItem>

### 2. ViewModifiers reutilizables

<Tabs syncKey="lang">
  <TabItem label="Swift">
    ```swift
    // Crear un modifier personalizado
    struct TarjetaModifier: ViewModifier {
        func body(content: Content) -> some View {
            content
                .padding()
                .background(Color(.systemBackground))
                .clipShape(.rect(cornerRadius: 12))
                .shadow(color: .black.opacity(0.1), radius: 5, y: 2)
        }
    }

    extension View {
        func estiloTarjeta() -> some View {
            modifier(TarjetaModifier())
        }

        .estiloTarjeta()
    ```

### 5. Inyección de dependencias

<Tabs syncKey="lang">
  <TabItem label="Swift">
    ```swift
    // Definir con protocolo
    @Observable
    class ProductosViewModel {
        private let repository: ProductoRepository
        var productos: [Producto] = []

        init(repository: ProductoRepository = APIProductoRepository()) {
            self.repository = repository
        }

        func cargar() async {
            do {
                productos = try await repository.obtenerTodos()
            } catch {
                print(error)
            }
        }
    }

    // En producción: usa API real
    let vm = ProductosViewModel()

    // En tests/previews: usa mock
    let vmTest = ProductosViewModel(repository: MockProductoRepository())
    ```
  </TabItem>
  <TabItem label="React">
    ```tsx
    "use client";

    import { createContext, useContext, ReactNode } from "react";

    // Context para inyectar la dependencia
    const ProductoRepoContext = createContext<ProductoRepository>(
      new APIProductoRepository()
    );

    function ProductoRepoProvider({
      repository,
      children,
    }: {
      repository: ProductoRepository;
      children: ReactNode;
    }) {
      return (
        <ProductoRepoContext.Provider value={repository}>
          {children}
        </ProductoRepoContext.Provider>
      );
    }

    function useProductoRepo() {
      return useContext(ProductoRepoContext);
    }

    // En producción
    <ProductoRepoProvider repository={new APIProductoRepository()}>
      <App />
    </ProductoRepoProvider>

    // En tests
    <ProductoRepoProvider repository={new MockProductoRepository()}>
      <App />
    </ProductoRepoProvider>
    ```
  </TabItem>
</Tabs>

:::caution
Evita usar singletons (`shared`) para todo. La inyección de dependencias hace tu código más testeable y flexible. Usa singletons solo para servicios verdaderamente globales (como la capa de red).
:::

## Checklist de un proyecto bien organizado

- [ ] Carpetas claras separando Models, Views, ViewModels, Services
- [ ] Una vista por archivo
- [ ] ViewModels no importan SwiftUI (solo `Foundation`)
- [ ] Componentes reutilizables en `Components/`
- [ ] Extensiones agrupadas en `Extensions/`
- [ ] Constantes centralizadas
- [ ] Previews con datos de ejemplo
- [ ] Sin lógica de negocio en las Views
- [ ] Protocolos para abstraer dependencias externas
- [ ] Nombres descriptivos y consistentes
