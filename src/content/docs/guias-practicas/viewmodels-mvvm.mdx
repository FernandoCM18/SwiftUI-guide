El patrón **MVVM** (Model-View-ViewModel) es la arquitectura más utilizada en apps SwiftUI. Separa la lógica de negocio (ViewModel) de la interfaz (View) y los datos (Model).

## Estructura MVVM

```
View       → Lo que el usuario ve (SwiftUI Views)
ViewModel  → Lógica de negocio y estado
Model      → Datos y estructuras
```

## ViewModel con @Observable (iOS 17+, recomendado)

Desde iOS 17, Apple recomienda usar el macro `@Observable` en lugar de `ObservableObject`:

```swift
import SwiftUI

@Observable
class ContadorViewModel {
    var valor = 0
    var historial: [Int] = []

    func incrementar() {
        valor += 1
        historial.append(valor)
    }

    func reiniciar() {
        valor = 0
        historial.removeAll()
    }
}

struct ContadorView: View {
    @State private var vm = ContadorViewModel()

    var body: some View {
        VStack(spacing: 20) {
            Text("\(vm.valor)")
                .font(.system(size: 72, weight: .bold))

            HStack(spacing: 16) {
                Button("Reiniciar") { vm.reiniciar() }
                    .buttonStyle(.bordered)

                Button("Sumar") { vm.incrementar() }
                    .buttonStyle(.borderedProminent)
            }

            if !vm.historial.isEmpty {
                Text("Historial: \(vm.historial.map(String.init).joined(separator: ", "))")
                    .font(.caption)
                    .foregroundStyle(.secondary)
            }
        }
    }
}
```

:::tip
Con `@Observable`, usa `@State` en la vista que crea el ViewModel. Es más simple que el enfoque anterior con `@StateObject`.
:::

## Compatibilidad legacy (iOS 14-16)

:::note
Si necesitas dar soporte a iOS 14-16, usa `ObservableObject` con `@Published` y `@StateObject`. En iOS 17+ usa siempre `@Observable`.
:::

```swift
// Solo para proyectos que soporten iOS 14-16
class TareasViewModel: ObservableObject {
    @Published var tareas: [String] = []
}

struct TareasView: View {
    @StateObject private var vm = TareasViewModel()
    var body: some View { Text("\(vm.tareas.count) tareas") }
}
```

:::note[Property wrappers legacy]
- `@StateObject` — vista que **crea** el ViewModel
- `@ObservedObject` — vista hija que **recibe** el ViewModel
- `@EnvironmentObject` — compartir ViewModel en toda la jerarquía
:::

## @Observable vs ObservableObject (referencia)

| Característica | `@Observable` (iOS 17+) | `ObservableObject` (iOS 14+) |
|---|---|---|
| Declaración | `@Observable class VM` | `class VM: ObservableObject` |
| Propiedades | Automáticas | Necesitan `@Published` |
| En la vista | `@State` | `@StateObject` |
| Rendimiento | Tracking granular por propiedad | Re-render por cualquier cambio |
| Complejidad | Menor | Mayor |

:::tip
Si tu app soporta iOS 17+, usa `@Observable`. Es más simple, más eficiente y requiere menos código boilerplate.
:::

## Ejemplo completo: App de notas con MVVM

### Model

```swift
struct Nota: Identifiable, Codable {
    let id: UUID
    var titulo: String
    var contenido: String
    var fecha: Date

    init(titulo: String, contenido: String) {
        self.id = UUID()
        self.titulo = titulo
        self.contenido = contenido
        self.fecha = Date()
    }
}
```

### ViewModel

```swift
@Observable
class NotasViewModel {
    var notas: [Nota] = []
    var notaSeleccionada: Nota?
    var busqueda = ""

    var notasFiltradas: [Nota] {
        if busqueda.isEmpty { return notas }
        return notas.filter {
            $0.titulo.localizedCaseInsensitiveContains(busqueda) ||
            $0.contenido.localizedCaseInsensitiveContains(busqueda)
        }
    }

    func agregar(titulo: String, contenido: String) {
        let nota = Nota(titulo: titulo, contenido: contenido)
        notas.insert(nota, at: 0)
    }

    func eliminar(at offsets: IndexSet) {
        notas.remove(atOffsets: offsets)
    }

    func actualizar(_ nota: Nota, titulo: String, contenido: String) {
        if let index = notas.firstIndex(where: { $0.id == nota.id }) {
            notas[index].titulo = titulo
            notas[index].contenido = contenido
        }
    }
}
```

### View

```swift
struct NotasAppView: View {
    @State private var vm = NotasViewModel()
    @State private var mostrarFormulario = false

    var body: some View {
        NavigationStack {
            List {
                ForEach(vm.notasFiltradas) { nota in
                    NavigationLink {
                        DetalleNotaView(vm: vm, nota: nota)
                    } label: {
                        VStack(alignment: .leading) {
                            Text(nota.titulo)
                                .font(.headline)
                            Text(nota.contenido)
                                .font(.caption)
                                .foregroundStyle(.secondary)
                                .lineLimit(1)
                        }
                    }
                }
                .onDelete { vm.eliminar(at: $0) }
            }
            .navigationTitle("Notas")
            .searchable(text: $vm.busqueda)
            .toolbar {
                Button { mostrarFormulario = true } label: {
                    Image(systemName: "plus")
                }
            }
            .sheet(isPresented: $mostrarFormulario) {
                NuevaNotaView(vm: vm)
            }
        }
    }
}
```
