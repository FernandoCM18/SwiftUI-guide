## Defining Interfaces and Contracts

Protocols define a blueprint of methods, properties, and other requirements that a type must implement. They are central to Swift's design philosophy of protocol-oriented programming.

## Declaring a Protocol

<Tabs syncKey="lang">
  <TabItem label="Swift">
    ```swift
    protocol Describable {
        var description: String { get }
        func describe()
    }

    // Conforming to a protocol
    struct Dog: Describable {
        var name: String

        var description: String {
            "Dog named \(name)"
        }

        func describe() {
            print(description)
        }
    }

    let dog = Dog(name: "Rex")
    dog.describe()  // "Dog named Rex"
    ```
  </TabItem>
  <TabItem label="React (Next.js)">
    ```typescript
    interface Describable {
      readonly description: string;
      describe(): void;
    }

    // Implementing an interface
    class Dog implements Describable {
      constructor(public name: string) {}

      get description(): string {
        return `Dog named ${this.name}`;
      }

      describe(): void {
        console.log(this.description);
      }
    }

    const dog = new Dog("Rex");
    dog.describe();  // "Dog named Rex"
    ```
  </TabItem>
</Tabs>

## Property Requirements

Protocols can require properties to be gettable or gettable and settable.

```swift
protocol Identifiable {
    var id: String { get }          // Read-only required
    var name: String { get set }    // Read-write required
}

struct User: Identifiable {
    let id: String         // let satisfies { get }
    var name: String       // var satisfies { get set }
}

var user = User(id: "001", name: "Alice")
print(user.id)      // "001"
user.name = "Bob"   // Allowed because name is { get set }
```

## Method Requirements

```swift
protocol Togglable {
    mutating func toggle()
}

enum Switch: Togglable {
    case on, off

    mutating func toggle() {
        self = (self == .on) ? .off : .on
    }
}

var lightSwitch = Switch.off
lightSwitch.toggle()
print(lightSwitch)  // on
lightSwitch.toggle()
print(lightSwitch)  // off
```

## Protocol Conformance with Multiple Protocols

<Tabs syncKey="lang">

    protocol Printable {
        func printDetails()
    }

    protocol Storable {
        func save()
    }

    struct Document: Printable, Storable {

        var content: String

            print("Document: \(title)")

            print("Saving '\(title)' to disk...")

## Protocol Inheritance

Protocols can inherit from other protocols.

<Tabs syncKey="lang">
  <TabItem label="Swift">
    ```swift
    protocol Named {
        var name: String { get }
    }

    protocol Aged {
        var age: Int { get }
    }

    // PeopleProtocol inherits both
    protocol PersonProtocol: Named, Aged {
        func greet() -> String
    }

    struct Person: PersonProtocol {
        var name: String
        var age: Int

        func greet() -> String {
            "Hi, I'm \(name), \(age) years old."
        }
    }

    let person = Person(name: "Alice", age: 30)
    print(person.greet())
    ```
  </TabItem>
  <TabItem label="React (Next.js)">
    ```typescript
    interface Named {
      readonly name: string;
    }

    interface Aged {
      readonly age: number;
    }

    // PersonInterface extends both
    interface PersonInterface extends Named, Aged {
      greet(): string;
    }

    const person: PersonInterface = {
      name: "Alice",
      age: 30,
      greet() {
        return `Hi, I'm ${this.name}, ${this.age} years old.`;
      },
    };

    console.log(person.greet());
    ```
  </TabItem>
</Tabs>

## Default Implementations with Extensions

Protocol extensions let you provide default implementations for any method or property.

<Tabs syncKey="lang">
  <TabItem label="Swift">
    ```swift
    protocol Greetable {
        var name: String { get }
        func greet() -> String
    }

    // Default implementation
    extension Greetable {
        func greet() -> String {
            "Hello, my name is \(name)."
        }
    }

    struct Student: Greetable {
        var name: String
        // Uses the default greet() implementation
    }

    struct Teacher: Greetable {
        var name: String

        // Provides a custom implementation
        func greet() -> String {
            "I'm Professor \(name)."
        }
    }

    let student = Student(name: "Alice")
    let teacher = Teacher(name: "Smith")

    print(student.greet())  // "Hello, my name is Alice."
    print(teacher.greet())  // "I'm Professor Smith."
    ```
  </TabItem>
  <TabItem label="React (Next.js)">
    ```typescript
    // TS has no direct equivalent to protocol extensions.
    // Use a utility function to provide default behavior.

    interface Greetable {
      readonly name: string;
      greet(): string;
    }

    function defaultGreet(name: string): string {
      return `Hello, my name is ${name}.`;
    }

    // Uses the default greeting via utility function
    const student: Greetable = {
      name: "Alice",
      greet() {
        return defaultGreet(this.name);
      },
    };

    // Custom implementation
    const teacher: Greetable = {
      name: "Smith",
      greet() {
        return `I'm Professor ${this.name}.`;
      },
    };

    console.log(student.greet());  // "Hello, my name is Alice."
    console.log(teacher.greet());  // "I'm Professor Smith."
    ```
  </TabItem>
</Tabs>

## Protocols as Types

You can use a protocol as a type for variables, parameters, and return values.

```swift
protocol Shape {
    var area: Double { get }
    func describe() -> String
}

struct Circle: Shape {
    var radius: Double
    var area: Double { .pi * radius * radius }
    func describe() -> String { "Circle with radius \(radius)" }
}

struct Square: Shape {
    var side: Double
    var area: Double { side * side }
    func describe() -> String { "Square with side \(side)" }
}

// Using protocol as a type
let shapes: [Shape] = [
    Circle(radius: 5),
    Square(side: 4),
    Circle(radius: 3)
]

for shape in shapes {
    print("\(shape.describe()) - Area: \(shape.area)")
}
```

## Protocol-Oriented Programming

Swift encourages building functionality through protocol composition rather than class inheritance.

<Tabs syncKey="lang">
  <TabItem label="Swift">
    ```swift
    protocol Flying {
        var maxAltitude: Double { get }
        func fly()
    }

    protocol Swimming {
        var maxDepth: Double { get }
        func swim()
    }

    extension Flying {
        func fly() {
            print("Flying up to \(maxAltitude) meters")
        }
    }

    extension Swimming {
        func swim() {
            print("Swimming down to \(maxDepth) meters")
        }
    }

    // A duck can both fly and swim
    struct Duck: Flying, Swimming {
        var maxAltitude: Double = 500
        var maxDepth: Double = 2

        var name: String
    }

    let duck = Duck(name: "Donald")
    duck.fly()   // "Flying up to 500.0 meters"
    duck.swim()  // "Swimming down to 2.0 meters"
    ```
  </TabItem>
  <TabItem label="React (Next.js)">
    ```typescript
    interface Flying {
      maxAltitude: number;
      fly(): void;
    }

    interface Swimming {
      maxDepth: number;
      swim(): void;
    }

    // Intersection type: must satisfy both interfaces
    type FlyingSwimmer = Flying & Swimming;

    const duck: FlyingSwimmer = {
      maxAltitude: 500,
      maxDepth: 2,
      fly() {
        console.log(`Flying up to ${this.maxAltitude} meters`);
      },
      swim() {
        console.log(`Swimming down to ${this.maxDepth} meters`);
      },
    };

    duck.fly();   // "Flying up to 500 meters"
    duck.swim();  // "Swimming down to 2 meters"
    ```
  </TabItem>
</Tabs>

:::tip
Prefer protocol composition over deep class hierarchies. It gives you more flexibility and avoids the problems of multiple inheritance.
:::

## Common Standard Library Protocols

```swift
// Equatable: allows == comparison
struct Point: Equatable {
    var x: Double
    var y: Double
}

let p1 = Point(x: 1, y: 2)
let p2 = Point(x: 1, y: 2)
print(p1 == p2)  // true

// Comparable: allows < > comparisons and sorting
struct Score: Comparable {
    var value: Int

    static func < (lhs: Score, rhs: Score) -> Bool {
        lhs.value < rhs.value
    }
}

let scores = [Score(value: 85), Score(value: 92), Score(value: 78)]
let sorted = scores.sorted()
print(sorted.map { $0.value })  // [78, 85, 92]

// CustomStringConvertible
struct Temperature: CustomStringConvertible {
    var celsius: Double

    var description: String {
        "\(celsius)°C"
    }
}

let temp = Temperature(celsius: 22.5)
print(temp)  // "22.5°C"
```
