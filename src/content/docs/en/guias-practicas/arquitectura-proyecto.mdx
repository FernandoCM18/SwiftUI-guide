## Naming conventions

### Files and types

<Tabs syncKey="lang">
  <TabItem label="Swift">
    ```swift
    // Views: "View" suffix
    struct ProductDetailView: View { }
    struct LoginView: View { }

    // ViewModels: "ViewModel" suffix
    class ProductsViewModel { }
    class AuthViewModel { }

    // Models: concept name
    struct Product { }
    struct User { }

    // Services: "Service" suffix
    class APIService { }
    class AuthService { }

    // Reusable components: descriptive
    struct PrimaryButton: View { }
    struct ProductCard: View { }
    ```
  </TabItem>
  <TabItem label="React">
    ```tsx
    // Pages: page.tsx inside route folder
    // app/products/[id]/page.tsx

    // Components: PascalCase, no suffix needed
    function ProductDetail() { }
    function LoginForm() { }

    // Hooks: "use" prefix
    function useProducts() { }
    function useAuth() { }

    // Types: concept name (PascalCase)
    interface Product { }
    interface User { }

    // Services / lib: camelCase files
    // lib/api.ts, lib/auth.ts

    // Reusable components: descriptive
    function PrimaryButton() { }
    function ProductCard() { }
    ```
  </TabItem>
</Tabs>

### Variables and functions

<Tabs syncKey="lang">
  <TabItem label="Swift">
    ```swift
    // Properties: nouns in camelCase
    var userName: String
    var isLoading: Bool
    var productList: [Product]

    // Functions: verbs in camelCase
    func loadData() { }
    func saveProfile() { }
    func deleteProduct(_ product: Product) { }

    // Closures/callbacks: "on" + action
    var onSave: () -> Void
    var onSelect: (Product) -> Void
    ```
  </TabItem>
  <TabItem label="React">
    ```typescript
    // State: nouns in camelCase
    const [userName, setUserName] = useState("");
    const [isLoading, setIsLoading] = useState(false);
    const [products, setProducts] = useState<Product[]>([]);

    // Functions: verbs in camelCase
    function loadData() { }
    function saveProfile() { }
    function deleteProduct(product: Product) { }

    // Callbacks (props): "on" + action
    interface Props {
      onSave: () => void;
      onSelect: (product: Product) => void;
    }
    ```
  </TabItem>

## Architecture patterns

### MVVM (recommended for SwiftUI)

<Tabs syncKey="lang">
  <TabItem label="Swift">
    ```swift
    // Model: pure data
    struct Task: Identifiable {
        let id = UUID()
        var title: String
        var completed: Bool
    }

    // ViewModel: business logic
    @Observable
    class TasksViewModel {
        var tasks: [Task] = []

        var pending: [Task] {
            tasks.filter { !$0.completed }
        }

        func add(_ title: String) {
            tasks.append(Task(title: title, completed: false))
        }

        func toggle(_ task: Task) {
            if let i = tasks.firstIndex(where: { $0.id == task.id }) {
                tasks[i].completed.toggle()
            }
        }
    }

    // View: UI only
    struct TasksView: View {
        @State private var vm = TasksViewModel()

        var body: some View {
            List(vm.tasks) { task in
                Text(task.title)
            }
        }
    }
    ```
  </TabItem>
  <TabItem label="React">
    ```tsx
    "use client";

    import { useState, useMemo, useCallback } from "react";

    // Type (replaces Model)
    interface Task {
      id: string;
      title: string;
      completed: boolean;
    }

    // Custom hook (replaces ViewModel)
    function useTasks() {
      const [tasks, setTasks] = useState<Task[]>([]);

      const pending = useMemo(
        () => tasks.filter((t) => !t.completed),
        [tasks]
      );

      const add = useCallback((title: string) => {
        setTasks((prev) => [
          ...prev,
          { id: crypto.randomUUID(), title, completed: false },
        ]);
      }, []);

      const toggle = useCallback((id: string) => {
        setTasks((prev) =>
          prev.map((t) =>
            t.id === id ? { ...t, completed: !t.completed } : t
          )
        );
      }, []);

      return { tasks, pending, add, toggle };
    }

    // Component (replaces View)
    export default function TasksPage() {
      const { tasks } = useTasks();

      return (
        <ul>
          {tasks.map((task) => (
            <li key={task.id}>{task.title}</li>
          ))}
        </ul>
      );
    }
    ```
  </TabItem>
</Tabs>

### Repository Pattern

Abstract the data source behind a protocol:

<Tabs syncKey="lang">
  <TabItem label="Swift">
    ```swift
    // Protocol
    protocol ProductRepository {
        func getAll() async throws -> [Product]
        func getById(_ id: Int) async throws -> Product
        func create(_ product: Product) async throws -> Product
    }

    // Real implementation (API)
    class APIProductRepository: ProductRepository {
        func getAll() async throws -> [Product] {
            try await APIService.shared.fetch("/products")
        }

        func getById(_ id: Int) async throws -> Product {
            try await APIService.shared.fetch("/products/\(id)")
        }

        func create(_ product: Product) async throws -> Product {
            try await APIService.shared.post("/products", body: product)
        }
    }

    // Mock implementation (for previews and tests)
    class MockProductRepository: ProductRepository {
        func getAll() async throws -> [Product] {
            [
                Product(id: 1, name: "iPhone", price: 999, inStock: true),
                Product(id: 2, name: "iPad", price: 799, inStock: false),
            ]
        }

        func getById(_ id: Int) async throws -> Product {
            Product(id: id, name: "Test", price: 0, inStock: true)
        }

        func create(_ product: Product) async throws -> Product {
            product
        }
    }
    ```
  </TabItem>
  <TabItem label="React">
    ```typescript
    // Interface (replaces Protocol)
    interface ProductRepository {
      getAll(): Promise<Product[]>;
      getById(id: number): Promise<Product>;
      create(product: Omit<Product, "id">): Promise<Product>;
    }

    // Real implementation (API)
    const apiProductRepo: ProductRepository = {
      getAll: () => apiFetch<Product[]>("/products"),
      getById: (id) => apiFetch<Product>(`/products/${id}`),
      create: (product) => apiPost<typeof product, Product>("/products", product),
    };

    // Mock implementation (for tests / Storybook)
    const mockProductRepo: ProductRepository = {
      getAll: async () => [
        { id: 1, name: "iPhone", price: 999, inStock: true },
        { id: 2, name: "iPad", price: 799, inStock: false },
      ],
      getById: async (id) => ({ id, name: "Test", price: 0, inStock: true }),
      create: async (product) => ({ id: 99, ...product }),
    };
    ```
  </TabItem>
</Tabs>

## Best practices

### 1. Small and composable views

<Tabs syncKey="lang">
  <TabItem label="Swift">
    ```swift
    // BAD: one giant view with everything
    struct HomeView: View {
        var body: some View {
            // 200+ lines of code...
        }
    }

    // GOOD: split into components
    struct HomeView: View {
        var body: some View {
            ScrollView {
                HeaderView()
                FeaturedProductsView()
                CategoriesView()
                PromoBannerView()
            }
        }
    }
    ```
  </TabItem>
  <TabItem label="React">
    ```tsx
    // BAD: one giant component with everything
    function HomePage() {
      return <main>{/* 200+ lines of JSX... */}</main>;
    }

    // GOOD: split into components
    function HomePage() {
      return (
        <main>
          <Header />
          <FeaturedProducts />
          <Categories />
          <PromoBanner />
        </main>
      );
    }
    ```
  </TabItem>

### 2. Reusable ViewModifiers

<Tabs syncKey="lang">
  <TabItem label="Swift">
    ```swift
    // Create a custom modifier
    struct CardModifier: ViewModifier {
        func body(content: Content) -> some View {
            content
                .padding()
                .background(Color(.systemBackground))
                .clipShape(.rect(cornerRadius: 12))
                .shadow(color: .black.opacity(0.1), radius: 5, y: 2)
        }
    }

    extension View {
        func cardStyle() -> some View {
            modifier(CardModifier())
        }
    }

    // Usage
    Text("Content")
        .cardStyle()
    ```
  </TabItem>
  <TabItem label="React">
    ```tsx
    // Wrapper component (replaces ViewModifier)
    function Card({ children }: { children: React.ReactNode }) {
      return (
        <div className="rounded-xl bg-white p-4 shadow-md dark:bg-gray-900">
          {children}
        </div>
      );
    }

    // Or use a Tailwind utility via cn() helper
    import { cn } from "@/lib/utils";

    const cardClass = "rounded-xl bg-white p-4 shadow-md dark:bg-gray-900";

    // Usage
    <Card>Content</Card>
    <div className={cn(cardClass, "mt-4")}>Content</div>
    ```

### 5. Dependency injection

<Tabs syncKey="lang">
  <TabItem label="Swift">
    ```swift
    // Define with protocol
    @Observable
    class ProductsViewModel {
        private let repository: ProductRepository
        var products: [Product] = []

        init(repository: ProductRepository = APIProductRepository()) {
            self.repository = repository
        }

        func load() async {
            do {
                products = try await repository.getAll()
            } catch {
                print(error)
            }
        }
    }

    // In production: uses real API
    let vm = ProductsViewModel()

    // In tests/previews: uses mock
    let vmTest = ProductsViewModel(repository: MockProductRepository())
    ```
  </TabItem>
  <TabItem label="React">
    ```tsx
    "use client";

    import { createContext, useContext, ReactNode } from "react";

    // Context for dependency injection
    const RepoContext = createContext<ProductRepository>(apiProductRepo);

    function RepoProvider({
      repo,
      children,
    }: {
      repo: ProductRepository;
      children: ReactNode;
    }) {
      return <RepoContext.Provider value={repo}>{children}</RepoContext.Provider>;
    }

    // Hook that reads the injected repository
    function useProducts() {
      const repo = useContext(RepoContext);
      // ... useState, useEffect with repo.getAll(), etc.
    }

    // In production
    <RepoProvider repo={apiProductRepo}><ProductsPage /></RepoProvider>

    // In tests / Storybook
    <RepoProvider repo={mockProductRepo}><ProductsPage /></RepoProvider>
    ```
  </TabItem>
</Tabs>

:::caution
Avoid using singletons (`shared`) for everything. Dependency injection makes your code more testable and flexible. Only use singletons for truly global services (like the networking layer).
:::

## Well-organized project checklist

- [ ] Clear folders separating Models, Views, ViewModels, Services
- [ ] One view per file
- [ ] ViewModels don't import SwiftUI (only `Foundation`)
- [ ] Reusable components in `Components/`
- [ ] Extensions grouped in `Extensions/`
- [ ] Centralized constants
- [ ] Previews with sample data
- [ ] No business logic in Views
- [ ] Protocols to abstract external dependencies
- [ ] Descriptive and consistent names
