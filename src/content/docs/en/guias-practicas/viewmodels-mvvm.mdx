The **MVVM** (Model-View-ViewModel) pattern is the most widely used architecture in SwiftUI apps. It separates business logic (ViewModel) from the interface (View) and data (Model).

## MVVM Structure

```
View       → What the user sees (SwiftUI Views)
ViewModel  → Business logic and state
Model      → Data and structures
```

## ViewModel with @Observable (iOS 17+, recommended)

Since iOS 17, Apple recommends using the `@Observable` macro instead of `ObservableObject`:

```swift
import SwiftUI

@Observable
class CounterViewModel {
    var value = 0
    var history: [Int] = []

    func increment() {
        value += 1
        history.append(value)
    }

    func reset() {
        value = 0
        history.removeAll()
    }
}

struct CounterView: View {
    @State private var vm = CounterViewModel()

    var body: some View {
        VStack(spacing: 20) {
            Text("\(vm.value)")
                .font(.system(size: 72, weight: .bold))

            HStack(spacing: 16) {
                Button("Reset") { vm.reset() }
                    .buttonStyle(.bordered)

                Button("Add") { vm.increment() }
                    .buttonStyle(.borderedProminent)
            }

            if !vm.history.isEmpty {
                Text("History: \(vm.history.map(String.init).joined(separator: ", "))")
                    .font(.caption)
                    .foregroundStyle(.secondary)
            }
        }
    }
}
```

:::tip
With `@Observable`, use `@State` in the view that creates the ViewModel. It's simpler than the previous approach with `@StateObject`.
:::

## Legacy compatibility (iOS 14-16)

:::note
If you need to support iOS 14-16, use `ObservableObject` with `@Published` and `@StateObject`. On iOS 17+ always use `@Observable`.
:::

```swift
// Only for projects supporting iOS 14-16
class TasksViewModel: ObservableObject {
    @Published var tasks: [String] = []
}

struct TasksView: View {
    @StateObject private var vm = TasksViewModel()
    var body: some View { Text("\(vm.tasks.count) tasks") }
}
```

:::note[Legacy property wrappers]
- `@StateObject` — view that **creates** the ViewModel
- `@ObservedObject` — child view that **receives** the ViewModel
- `@EnvironmentObject` — share ViewModel across the entire hierarchy
:::

## @Observable vs ObservableObject (reference)

| Feature | `@Observable` (iOS 17+) | `ObservableObject` (iOS 14+) |
|---|---|---|
| Declaration | `@Observable class VM` | `class VM: ObservableObject` |
| Properties | Automatic | Need `@Published` |
| In the view | `@State` | `@StateObject` |
| Performance | Granular per-property tracking | Re-render on any change |
| Complexity | Lower | Higher |

:::tip
If your app supports iOS 17+, use `@Observable`. It's simpler, more efficient, and requires less boilerplate code.
:::

## Complete example: Notes app with MVVM

### Model

```swift
struct Note: Identifiable, Codable {
    let id: UUID
    var title: String
    var content: String
    var date: Date

    init(title: String, content: String) {
        self.id = UUID()
        self.title = title
        self.content = content
        self.date = Date()
    }
}
```

### ViewModel

```swift
@Observable
class NotesViewModel {
    var notes: [Note] = []
    var selectedNote: Note?
    var searchText = ""

    var filteredNotes: [Note] {
        if searchText.isEmpty { return notes }
        return notes.filter {
            $0.title.localizedCaseInsensitiveContains(searchText) ||
            $0.content.localizedCaseInsensitiveContains(searchText)
        }
    }

    func add(title: String, content: String) {
        let note = Note(title: title, content: content)
        notes.insert(note, at: 0)
    }

    func delete(at offsets: IndexSet) {
        notes.remove(atOffsets: offsets)
    }

    func update(_ note: Note, title: String, content: String) {
        if let index = notes.firstIndex(where: { $0.id == note.id }) {
            notes[index].title = title
            notes[index].content = content
        }
    }
}
```

### View

```swift
struct NotesAppView: View {
    @State private var vm = NotesViewModel()
    @State private var showForm = false

    var body: some View {
        NavigationStack {
            List {
                ForEach(vm.filteredNotes) { note in
                    NavigationLink {
                        NoteDetailView(vm: vm, note: note)
                    } label: {
                        VStack(alignment: .leading) {
                            Text(note.title)
                                .font(.headline)
                            Text(note.content)
                                .font(.caption)
                                .foregroundStyle(.secondary)
                                .lineLimit(1)
                        }
                    }
                }
                .onDelete { vm.delete(at: $0) }
            }
            .navigationTitle("Notes")
            .searchable(text: $vm.searchText)
            .toolbar {
                Button { showForm = true } label: {
                    Image(systemName: "plus")
                }
            }
            .sheet(isPresented: $showForm) {
                NewNoteView(vm: vm)
            }
        }
    }
}
```
